-- формат команд: название команды ключи аргументы
-- ключи - то, как команде надо работать
-- аргументы - то, с чем команде надо работать


------------------------------cd change directory-----------------------
~ - домашняя директория
cd / - выйти в корневой каталог
cd без аргументов - попадем в домашнюю директорию
cd - - возвращает в предыдущий каталог


------------------------------cp-----------------------------------------
копирование файла
cp -r - для директории

------------------------------cut-----------------------------------------
вырезать 
cut -c2-10 - вырезать символы со 2 по 10


------------------------------wc word count-----------------------------------------

wc -l - количество строк



------------------------------------------------id----------------------
информация о пользователе
id root - информация о пользователе


-------------------------------less навигация по файлу------------------
less file


-------------------------------ls----------------------------------------

ls -la - длинный вывод с показом всех скрытых файлов
ls -ld - узнать информацию о директории



--------------------------------mkdir--------------------------------------
создание каталога



--------------------------------mv-----------------------------------------
перемещение или переименовывание



--------------------------pwd - print work directory----------------------




----------------------------------rm---------------------------------------
удаление файла




----------------------------------rmdir------------------------------------
удаление директории
rm -r - рекурсивное удаление директории со всем её содержимым
rm -rf - удалить рекурсивно и без вопросов


--------------------------------tail--------------------------------------
tail -n 15 file - выводит последние 15 строк файла 

--------------------------------touch--------------------------------------
создание файла

command_1 && command_2 - command_2 выполнится только в том случае, если успешно выполнилась первая команда
command_1 || command_2 - command_2 выполнится только в том случае, если первая выполнилась с ошибкой
command_1 ; command_2 - command_2 выполнять следующую команду независимо от того, как выполнилась предыдущая

#!/bin/bash - указываем к какому командному интерпретатору относится дальнейший код инструкций

#!/bin/bash

for y in {2017..2020}
do
        mkdir $y
        for m in {01..12}
        do
                mkdir $y/$m
                for f in {1..10}
                do
                        echo file$f.txt >> $y/$m/file$f.txt
                done
        done
done
